# Classwork 12 - The Mandelbrot Set (Visualization)
# Pseudocode (PPP)

START

# Step 1: read the configuration file into a dictionary
CREATE empty dictionary config
OPEN file "config.txt" for reading
READ all lines from file into a list
FOR each line in the list
    SPLIT line at "=" into parameter and value
    IF value contains a decimal point THEN
        config at parameter <- value converted to decimal number
    ELSE
        config at parameter <- value converted to whole number
    END IF
END FOR
CLOSE file

# Step 2: extract the values needed for the image
max_iterations <- config value for "max_iter"
width <- config value for "ancho"
height <- config value for "alto"

# Step 3: read the previously saved Mandelbrot data
OPEN file "mandelbrot.csv" for reading
READ all lines from file into a list
CLOSE file
REMOVE the first line from the list (it is the header, not data)

# Step 4: create a blank image, one pixel per grid point
CREATE new image using HSV color mode, with size (width, height)

# Step 5: go through every saved data point
FOR each remaining line in the list
    SPLIT line at "," into row, column, iterations
    CONVERT row, column, iterations to whole numbers

    # Step 6: turn the iteration count into a brightness value
    IF iterations equals max_iterations THEN
        brightness <- 0                       # point never escaped -> inside the set
    ELSE
        brightness <- (iterations / max_iterations) * 255   # scale to 0-255
        CONVERT brightness to a whole number
    END IF

    # Step 7: color this pixel using the brightness as the HSV value channel
    SET pixel at (column, row) <- (brightness, 255, 255)

END FOR

# Step 8: convert the image to standard RGB color and save it
CONVERT image from HSV to RGB
SAVE image as "mandelbrot.png"

DISPLAY "DONE"

END
