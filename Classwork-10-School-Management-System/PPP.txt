INPUT
   Get usuario, contraseña from user

PROCESS
   # Step 1: authenticate the user (repeat until valid credentials)
   SET acceso <- False
   LOOP while acceso = False:
      IF usuario exists in usuarios AND usuarios[usuario]['password'] = contraseña:
         SET acceso <- True
      ELSE:
         Display "Usuario o contraseña incorrectos"

   nombre <- usuarios[usuario]['nombre']

   # Step 3: branch behavior according to role

      SET aprobadas <- empty set
      LOOP for each materia in materias:
         calif <- calificaciones[usuario][materia]

      pendientes <- materias - aprobadas

            Get alumno from user

            Display "Esa materia no existe."
            Get materia from user

OUTPUT
   Display "Bienvenido, " + nombre + " (" + rol + ")"
   IF rol = "alumno": Display calif per materia, then aprobadas, pendientes
